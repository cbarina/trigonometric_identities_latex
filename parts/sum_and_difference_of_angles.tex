\subsection*{Sum and difference of angles}

\begin{align*}
  \sin(\theta + y) &= \sin \theta \cos y + \cos \theta \sin y\\
  \sin(\theta - y) &= \sin \theta \cos y - \cos \theta \sin y\\
  \cos(\theta + y) &= \cos \theta \cos y - \sin \theta \sin y\\
  \cos(\theta - y) &= \cos \theta \cos y + \sin \theta \sin y\\
  \tan(\theta + y) &= \frac{\tan \theta + \tan y}{1 - \tan \theta \tan y}\\
  \tan(\theta - y) &= \frac{\tan \theta - \tan y}{1 + \tan \theta \tan y}
\end{align*}
